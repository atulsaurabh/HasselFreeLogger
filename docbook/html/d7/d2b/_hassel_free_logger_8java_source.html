311  /*
312  I believe that all kind of log method requires level to decide
313  what kind of file need to be generated and what format to use while
314  recording the log. So this decision can be made centrally. That is why I
315  designed this central method to produce a logger object with the decision of
316  a) The level of the log
317  b) which class is generating the log
318 
319  That is why this method requies two parameters
320  a) errorClass : where the exception is generated
321  b) level of the log.
322  */
323  private org.apache.log4j.Logger log(Class errorClass,Level level)
324  {
325  /*
326  If the user of the system do not want to supply error generating class
327  then by default the current class will be considered to be the class which is
328  generating the eeror.
329  */
330  org.apache.log4j.Logger logger=null;
331  if(errorClass == null)
332  logger= org.apache.log4j.Logger.getLogger(this.getClass().getName());
333  else
334  /*
335 
336  */
337  logger=org.apache.log4j.Logger.getLogger(errorClass.getName());
338  /*
339  If the user want to record the log into respective file then the rollingOn
340  should be true.
341  */
342  if(isRollingOn())
343  {
344  try {
345  /*
346  I want to record different level of log into different files.
347  So what file wiil be used will depened upon the level.
348  */
349  String fileName="default.log";
350  if(level == Level.ALL)
351  fileName = all_log_file_name;
352  else if(level == Level.DEBUG)
353  fileName = debug_log_file_name;
354  else if(level == Level.ERROR)
355  fileName = error_log_file_name;
356  else if(level == Level.FATAL)
357  fileName = fatal_log_file_name;
358  else if(level == Level.INFO)
359  fileName = info_log_file_name;
360  else if(level == Level.WARN)
361  fileName = warn_log_file_name;
362 
363  logger.addAppender(getRollingFileAdapter(fileName));
364  } catch (URISyntaxException e) {
365  java.util.logging.Logger.getLogger(this.getClass().getName()).log(java.util.logging.Level.SEVERE,
366  "URL Is Not Valid For Log File\nUnable To Create Daily LOG FILE");
367  logger.addAppender(new ConsoleAppender());
368  }
369  }
370  logger.setLevel(level);
371  return logger;
372  }
380  private DailyRollingFileAppender getRollingFileAdapter(String fileName) throws URISyntaxException
381  {
382  URI logDIRFullName = this.getClass().getResource("/").toURI();
383  String normalizePath = Paths.get(logDIRFullName).toString();
384 
385  String log_file_full_name = normalizePath + "/" + log_directory_name + "/" + fileName;
386  PatternLayout layout = new PatternLayout();
387  String conversionPattern = getConversionPattern();
388  layout.setConversionPattern(getConversionPattern());
389  DailyRollingFileAppender rollingAppender = new DailyRollingFileAppender();
390  rollingAppender.setFile(log_file_full_name);
391  rollingAppender.setDatePattern("'.'"+getDatePattern());
392  rollingAppender.setLayout(layout);
393  rollingAppender.activateOptions();
394  return rollingAppender;
395  }
